RewriteEngine On
RewriteBase /

# 1. Redirect dari URL yang masih pakai .html ke URL tanpa .html
RewriteCond %{THE_REQUEST} \s/([^.]+)\.html[\s?] [NC]
RewriteRule ^ %1 [R=301,L]

# 2. Jika URL tanpa .html, tetap layani file .html
RewriteCond %{REQUEST_FILENAME}.html -f
RewriteRule ^([^/]+)/?$ $1.html [L]
